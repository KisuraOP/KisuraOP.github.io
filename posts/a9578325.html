A. 找矩阵

题意：给定一个由 . # S 三种字符构成的 \(n\times m\) 矩阵，判断是否存在一个长宽均至少为 \(2\) 的矩形方框，使其经过 S 而不经过 #。保证字符 S 只出现一次。

\(1 \le n,m \le 3 \cdot 10^3\)。

法一（枚举+二维前缀和）：

先考虑起点 S 在目标矩形的边上的情况。此时从 S 出发绕一圈回到 S 共有四种方案，如下图：

其中划绿色的部分的长度需要枚举，其余部分则可以用二维前缀和 \(O(1)\) 判断是否合法。

至于起点 S 在目标矩形角上的情况，其实就是把 S 所在的两段边并到最后枚举了，细节处理好一点就能被第一种情况囊括。

void solve()
{
	int n = read(), m = read() + 1;
	vector<string> s(n + 2);
	int sx = 0, sy = 0;
	for (int i = 1; i <= n; i++) {
		cin >> s[i];
		s[i] = ' ' + s[i] + '#';
		for (int j = 1; j <= m; j++) {
			if (s[i][j] == 'S') sx = i, sy = j;
		}
	}
	s[++n] = string('#', m + 2);
	vector<vector<int>> sum(n + 1, vector<int>(m + 1, 0));
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= m; j++) {
			sum[i][j] = sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1] + (s[i][j] == '#');
		}
	}
	auto isok = [&] (int x1, int y1, int x2, int y2) {
		int res = sum[x2][y2] - sum[x1 - 1][y2] - sum[x2][y1 - 1] + sum[x1 - 1][y1 - 1];
		return res == 0;
	};
	
	for (int left = sy; left >= 1 && s[sx][left] != '#'; left--) {
		for (int down = sx + 1; down <= n && s[down][left] != '#'; down++) {
			if (isok(down, left, down, sy)) {
				for (int right = sy + 1; right <= m && s[down][right] != '#'; right++) {
					if (isok(sx, right, down, right) && isok(sx, sy, sx, right)) die("Yes");
				}
			}
		}
	}
	for (int left = sy; left >= 1 && s[sx][left] != '#'; left--) {
		for (int up = sx - 1; up >= 1 && s[up][left] != '#'; up--) {
			if (isok(up, left, up, sy)) {
				for (int right = sy + 1; right <= m && s[up][right] != '#'; right++) {
					if (isok(up, right, sx, right) && isok(sx, sy, sx, right)) die("Yes");
				}
			}
		}
	}
	for (int up = sx; up >= 1 && s[up][sy] != '#'; up--) {
		for (int left = sy - 1; left >= 1 && s[up][left] != '#'; left--) {
			if (isok(up, left, sx, left)) {
				for (int down = sx + 1; down <= n && s[down][left] != '#'; down++) {
					if (isok(down, left, down, sy) && isok(sx, sy, down, sy)) die("Yes");
				}
			}
		}
	}
	for (int up = sx; up >= 1 && s[up][sy] != '#'; up--) {
		for (int right = sy + 1; right <= m && s[up][right] != '#'; right++) {
			if (isok(up, right, sx, right)) {
				for (int down = sx + 1; down <= n && s[down][right] != '#'; down++) {
					if (isok(down, sy, down, right) && isok(sx, sy, down, sy)) die("Yes");
				}
			}
		}
	}
	die("No");
}

法二（\(\text{bfs}\)）：

钦定初始的方向跑 \(4\) 遍 \(\text{bfs}\)。在一次 \(\text{bfs}\) 中，枚举每个点是否转向，而转向的方向也可以钦定为顺时针或逆时针，判断在 \(4\) 次转向内能否回到原点即可。

提供一种代码简洁的写法。

两种方法时间复杂度均为 \(O(mn)\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in", "r", stdin);freopen(#x".out", "w", stdout)
#define ck(x) printf("check %lld\n", x);fflush(stdout);
#define die(x) { puts(x);return ;}
#define int long long
#define double long double
#define inf 0x3fffffffffffffff
inline int read()
{
	int x = 0, f = 1; char ch = getchar();
	while (!isdigit(ch)) { if (ch == '-') f = -1; ch = getchar();}
	while (isdigit(ch)) { x = x * 10 + ch - 48; ch = getchar();}
	return f == 1 ? x : -x;
}
inline void write(int x)
{
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) write(x / 10);
    putchar(x % 10 + '0');
}


void solve()
{
	int n = read(), m = read();
	vector<string> s(n);
	int sx, sy;
	for (int i = 0; i < n; i++) {
		cin >> s[i];
		auto it = s[i].find('S');
		if (it != -1) {
			sx = i, sy = it;
		}
	}
	int dx[] = {-1, 0, 1, 0};
	int dy[] = {0, 1, 0, -1};
	int limit[] = {(n + m) * 2, 4};
	for (int f = 0; f < 4; f++) {
		queue<array<int, 3>> q;
		vector<vector<bool>> vis(n, vector<bool>(m));
		q.push({sx, sy, 0});
		while (!q.empty()) {
			auto [x, y, cnt] = q.front();
			q.pop();
			if (vis[x][y]) continue;
			vis[x][y] = true;
			for (auto dir : {0, 1}) {
				if (cnt + 1 > limit[dir]) continue;
				int xx = x + dx[(f + cnt + dir) % 4];
				int yy = y + dy[(f + cnt + dir) % 4];
				if (xx < 0 || yy < 0 || xx >= n || yy >= m) continue;
				if (xx == sx && yy == sy) die("Yes");
				if (vis[xx][yy] || s[xx][yy] == '#') continue;
				q.push({xx, yy, cnt + dir});
			}
		}
 	}	
 	die("No");
}


signed main()
{
	fre(test);
	int T = 1;
	while (T--) {
		solve();
		//fflush(stdout);
	}
	return 0;
}

B. 错峰旅行

题意：有 \(n\) 个互相连通的城市，给定 \(m\) 条信息每条形如城市 \(x\) 在第 \(l\sim r\) 天是拥挤的。旅行者每天只会在一个不拥挤的城市停留，第二天可以选择到达任何一个不拥挤的城市（包括当前城市）。他的旅游从第 \(S\) 天至第 \(T\) 天，不要求逛完所有城市且能从任何一个城市开始，问合法旅游方案数。对 \(10^9+7\) 取模。

\(1\le n\le 5000\)，\(1\le m \le 10^6\)，\(0\le S \le l,r \le T \le 10^9\)。

C. 社交树

题意：给定 \(n\) 个点的树，维护 \(m\) 条操作：

  • 1 v x，给 \(v\) 的子树中每个点 \(u\) 加上 \(a^{x+d}\)，\(d\) 为 \(u,v\) 间的深度差。
  • 2 v，询问 \(v\) 号点的权值。\(\pmod {10^8+7}\)

\(1\le n,m \le 10^5\)，\(0 < |a|,|x| \le 10^3\)。

负数取模要化为正数。

F. 图论大师

题意：给定 \(n\) 个点 \(m\) 条边的无向图，边有边权 \(w_i\)。

定义两点之间长度为路径（不一定是简单路径）上的边数，权值为路径上的权值和。

回答 \(q\) 次询问：对于所有从 \(u\to v\) 的长度为 \(k\) 的路径，权值的上下界是？

\(1\le n \le 100\)，\(1\le m\le 10^5\)，\(1\le q\le 20\)，\(1\le w_i \le 10^6\)，\(1 \le k_i \le 10^9\)。

H. 小度的双色球

题意：有 \(n\) 个盒子，第 \(i\) 个盒子装 \(a_i\) 个小球，其中第 \(j\) 个颜色为 \(c_{i,j}\)。

询问 \(Q\) 次，每次询问两个颜色 \(x,y(x\neq y)\)，问若每次等概率的从这 \(n\) 个盒子中抽一个，抽到同时含有颜色 \(x\) 和颜色 \(y\) 的小球的盒子的期望次数是？\(\pmod{998244353}\)

被抽中的盒子还会参与下次抽取。若永远无法抽到则输出 \(-1\)。

\(1 \le n,Q,c_{i,j},\sum a_i \le 2 \cdot 10^5\)。

J. 石子游戏

题意：给定若干堆石子，\(\text{A,B}\) 两人轮流行动，\(\text{A}\) 先手，规则是：

  1. 若全部石子被拿空，当前玩家输，游戏结束。
  2. 否则当前玩家需要从石子数最多的那堆拿走石子，至少一个，至多拿空。

现有一个可重集 \(S\)，初始为空，需要你维护以下操作：

  • 1 x y：向 \(S\) 中加入 \(y\) 个 \(x\)。
  • 2 l r：现生成一个集合 \(S'\)，对每个满足 \(x\in S,x\in[l,r]\) 的 \(x\)，有 \(\frac{1}{2}\) 的概率将其加入 \(S'\) 中。

对每个第二种操作，查询在此概率分布下，以 \(S'\) 为各堆石子的个数集，先手具有必胜策略的概率。\(\pmod {998244353}\)

\(1 \le n \le 5\cdot 10^5\)，\(1 \le x,y \le 10^9\)，\(1 \le l \le r \le 10^9\)。
