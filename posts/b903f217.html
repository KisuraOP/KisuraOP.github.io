A. Sorting with Twos

Problem

按题意模拟。

每个 \([2^i,2^{i+1}]\) 区间内若有递减，一定不能满足要求。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define int long long
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
inline int read()
{
	int x=0,f=1;char ch=getchar();
	while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
	while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
	return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 998244353;


void solve()
{
	int n=read();
	vector<int> a(n+1);
	for(int i=1;i<=n;++i) {
		a[i]=read();
	}
	int l=1,r=2;
	while(r<n) {
		for(int i=l+2;i<=r;++i) {
			if(a[i]<a[i-1]) {
				puts("No");
				return ;
			}
		}
		l*=2;
		r*=2;
	}
	for(int i=l+2;i<=min(r,n);++i) {
		if(a[i]<a[i-1]) {
			puts("No");
			return ;
		}
	}
	puts("Yes");
}


signed main()
{
	//fre(test);
	int T=read();
	while(T--) {
		solve();
		//fflush(stdout);
	}
	return 0;
}

B. Deja Vu

Problem

题意：给定长为 \(n\) 的正整数数列 \(a\) 和长为 \(q\) 的正整数数列 \(x\)。有 \(q\) 次操作。第 \(i\in[1,q]\) 次操作，对每个 \(j\in[1,n]\)，若 \(2^{x_i} \mid a_j\)，则 \(a_j \leftarrow a_j+2^{x_i-1}\)。输出最终数列。

\(1\le n,q \le 2\times 10^5\)，\(1\le a_i \le 10^{9}\)，\(1\le x_i \le 30\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
inline int read()
{
	int x=0,f=1;char ch=getchar();
	while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
	while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
	return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 998244353;


void solve()
{
	int n=read(),q=read();
	vector<int> a(n+1);
	vector<vector<int>> pos(31);
	for(int i=1;i<=n;++i) {
		a[i]=read();
		pos[__lg(a[i] & -a[i])].push_back(i);
	}
	for(int _=1;_<=q;++_) {
		int x=read();
		for(int i=x;i<=30;++i) {
			for(auto k:pos[i]) {
				a[k] += 1<<(x-1);
				pos[x-1].push_back(k);
			}
			pos[i].clear();
		}
	}
	for(int i=1;i<=n;++i) {
		printf("%lld ",a[i]);
	}
	puts("");
}


signed main()
{
	//fre(test);
	int T=read();
	while(T--) {
		solve();
		//fflush(stdout);
	}
	return 0;
}

一个数二进制最低位的 \(1\) 在第 \(m\) 位，那么其 \(\text{lowbit}\) 值为 \(2^{m}\)。

故数 \(x\) 的最低位 \(1\) 所在位置为：__lg(x & -x) 。

C. Smilo and Monsters

Problem

题意：有 \(n\) 组怪物，每组有 \(a_i\) 个，每次你可以选择一个组，选择杀死一只怪物，累积一个连击数或杀死连击数只怪物，连击数归零。问击败所有怪物所需最少攻击次数。

\(1\le n \le 2\times 10^5\)，\(1\le a_i \le 10^9\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define int long long
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
inline int read()
{
	int x=0,f=1;char ch=getchar();
	while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
	while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
	return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 998244353;


void solve()
{
	int n=read();
	vector<int> a(n+1);
	int sum=0;
	for(int i=1;i<=n;++i) {
		a[i]=read();
		sum+=a[i];
	}
	int ans=(sum+1)/2;
	sum/=2;
	int extra=0;
	while(sum>0) {
		sort(a.begin()+1,a.end(),greater<>());
		for(int i=1;i<=n;++i) {
			sum-=a[i];
			if(sum<=0) {
				extra=i;
				break;
			}
		}
	}
	printf("%lld\n",ans+extra);
}


signed main()
{
	//fre(test);
	int T=read();
	while(T--) {
		solve();
		// fflush(stdout);
	}
	return 0;
}

D. Suspicious logarithms

Problem

题意：定义 \(f(x)=\lfloor\log_{2}(x)\rfloor\)，\(g(x)=\lfloor \log_{f(x)}(x)\rfloor\)。

有 \(q\) 次查询，每次询问一个区间 \([l,r]\)，回答 \(\sum\limits_{i=l}^{r} g(i)\)。

\(1\le q\ \le 10^5\)，\(4\le l_i\le r_i \le 10^{18}\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define int unsigned long long
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
#define ck(x) printf("check %lld\n",x);fflush(stdout);
inline int read()
{
    int x=0,f=1;char ch=getchar();
    while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
    while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
    return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 1e9+7;

inline int g(int x,int fx) {
    int tot=-1,res=1;
    while(res<=x) {
        res*=fx;
        tot++;
    }
    return tot;
}

inline int qpow(int k,int n) {
    int s=1;
    for(;n;n>>=1,k=k*k) if(n&1) s=s*k;
    return s;
}


int calc(int n) {
    if(n<4) {
        return 0;
    }
    int ans=0;
    for(int bit=2;bit<=60;++bit) {
        int l=1ull<<bit;
        int r=min(n,2*l-1);
        int lans=g(l,bit);  
        int rans=g(r,bit);
        if(lans==rans) {
            ans+=(r-l+1)*lans;
        } else {
            int lst=l-1;
            for(int i=lans;i<=rans;++i) {
                int mid=qpow(bit,i+1)-1;
                mid=min(mid,r);
                ans+=(mid-lst)*i;
                lst=mid;
            }
        }
        if(r==n) {
            break;
        }
    }
    return ans;
}

void solve()
{
    int l=read(),r=read();
    printf("%lld\n",(calc(r)-calc(l-1))%modp);
}


signed main()
{
    //fre(test);
    int T=read();
    while(T--) {
        solve();
        fflush(stdout);
    }
    return 0;
}

E. Brukhovich and Exams

Problem

题意：给定长为 \(n\) 的非负整数序列 \(a\) 和正整数 \(k\)。定义 \(\text{sadness}\) 为序列中相邻两数互质的组数，你可以将最多 \(k\) 个 \(a_i\) 变为 \(0\)，问最小的 \(\text{sadness}\)。

注意对所有非负整数 \(x\)，有 \(\gcd(0,x)=\gcd(x,0)=x\)。

\(1\le k \le n \le 10^5\)，\(0\le a_i \le 10^9\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define int long long
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
#define ck(x) printf("check %lld\n",x);fflush(stdout);
inline int read()
{
	int x=0,f=1;char ch=getchar();
	while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
	while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
	return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 998244353;


void solve()
{
	int n=read(),k=read();
	vector<int> a(n+1);
	int all=0;
	bool all_one=true;
	for(int i=1;i<=n;++i) {
		a[i]=read();
		all_one&=(a[i]==1);
		if(i>1 && __gcd(a[i-1],a[i])==1) {
			all++;
		}
	}
	if(all_one) {
		printf("%lld\n",all-(k-1));
		return ;
	}

	vector<int> one,other;
	vector<vector<int>> block(1);
	block[0].push_back(1);
	int tot=0;
	if(a[1]==1) {
		one.push_back(tot);
	} else {
		other.push_back(tot);
	}
	for(int i=2;i<=n;++i) {
		if(a[i]==1) {
			if(a[i-1]==1) {
				block[tot].push_back(i);
			} else {
				block.push_back({});
				block[++tot].push_back(i);
				one.push_back(tot);
			}
		} else if(__gcd(a[i-1],a[i])==1 && a[i-1]!=1) {
			block[tot].push_back(i);
		} else {
			block.push_back({});
			block[++tot].push_back(i);
			other.push_back(tot);
		}
	}

	multiset<int> edge_one;
	if(find(one.begin(),one.end(),0)!=one.end()) {
		edge_one.insert(block[0].size());
		one.erase(find(one.begin(),one.end(),0));
	}
	if(tot && find(one.begin(),one.end(),tot)!=one.end()) {
		edge_one.insert(block[tot].size());
		one.erase(find(one.begin(),one.end(),tot));
	}

	multiset<int,greater<int>> q;
	for(auto i:other) {
		q.insert(block[i].size());
	} 
	while(!q.empty() && *q.begin()>=3 && k) {
		int i=*q.begin();
		if(i>=3) {
			q.erase(q.find(i));
			q.insert(i-2);
			all-=2;
			k--;
		}
	}
	
	multiset<int> p;
	for(auto i:one) {
		p.insert(block[i].size());
	}
	while(!p.empty()) {
		int i=*p.begin();
		if(i>k) {
			break;
		}
		p.erase(p.find(i));
		all-=i+1;
		k-=i;
	}
	
	if(k) {
		all-=k;
		k=0;
	}
	printf("%lld\n",max(0ll,all));
 }


signed main()
{
	//fre(test);
	int T=read();
	while(T--) {
		solve();
		//fflush(stdout);
	}
	return 0;
}

F. A Growing Tree

Problem

题意：给定一颗树，根为 \(1\)，初始只有根节点，每个节点有权值，初始为 \(0\)。两种操作共 \(q\) 次：

  • 1 v ：添加编号为 \(sz+1\)，权值为 \(0\) 的节点作为儿子挂在节点 \(v\) 下，\(sz\) 为当前节点个数。
  • 2 v x ：给编号为 \(v\) 的节点及其子树的左右节点的权值 \(+x\)。

问最终所有节点权值。

\(1\le q \le 5\times 10^5\)，\(10^{-9} \le x_i\le10^9\)。

Code 
#include <bits/stdc++.h>
using namespace std;
#define fre(x) freopen(#x".in","r",stdin);freopen(#x".out","w",stdout)
#define int long long
#define double long double
#define PI acos(-1)
#define inf 0x3fffffffffffffff
inline int read()
{
    int x=0,f=1;char ch=getchar();
    while(!isdigit(ch)){if(ch=='-')f=-1;ch=getchar();}
    while(isdigit(ch)){x=x*10+ch-48;ch=getchar();}
    return f==1?x:-x;
}
inline void write(int x)
{
    if(x<0) putchar('-'),x=-x;
    if(x>9) write(x/10);
    putchar(x%10+'0');
}
const int modp = 998244353;

struct SegmentTree {
    int n;
    vector<int> tree;
    vector<int> lazy_add;
    SegmentTree(int size) {
        n = size + 1;
        tree.resize(4*n+1,0);
        lazy_add.resize(4*n+1,0);
    }
    inline void pushup(int p) {
        tree[p] = tree[p<<1] + tree[p<<1|1];
    }
    inline void pushdown(int p,int left,int right) {
        if(lazy_add[p]) {
            int mid = left + right >> 1;
            lazy_add[p<<1]   += lazy_add[p];
            lazy_add[p<<1|1] += lazy_add[p];            
            tree[p<<1]       += (mid-left+1) * lazy_add[p];
            tree[p<<1|1]     += (right-mid)  * lazy_add[p];
            lazy_add[p] = 0;
        }
    }
    void in_build(int p,int left,int right,vector<int> &nums) {
        if(left == right) {
            tree[p] = nums[left];
        } else {
            int mid = left + right >> 1;
            in_build(p<<1,left,mid,nums);
            in_build(p<<1|1,mid+1,right,nums);
            pushup(p);
        }
    }
    void submit_change(int p, int left, int right, vector<int> &nums) {
        if (left == right) {
            nums[left] = tree[p];
        } else {
            pushdown(p,left,right);
            int mid = left + right >> 1;
            submit_change(p<<1,left,mid,nums);
            submit_change(p<<1|1,mid+1,right,nums);
            pushup(p);
        }
    }
    void update_add(int p,int left,int right,int ql,int qr,int val) {
        if(ql <= left && qr >= right) {
            tree[p]     += val * (right-left+1);
            lazy_add[p] += val;
        } else {
            pushdown(p,left,right);
            int mid = left + right >> 1;
            if(ql <= mid) update_add(p<<1,left,mid,ql,qr,val);
            if(qr > mid)  update_add(p<<1|1,mid+1,right,ql,qr,val);
            pushup(p);
        }
    }
    void update_change(int p,int left,int right,int ql,int qr,int val) {
        if (ql <= left && qr >= right) {
            tree[p]     = val * (right - left + 1);
            lazy_add[p] = val;
        } else {
            pushdown(p,left,right);
            int mid = left + right >> 1;
            if (ql <= mid) update_change(p<<1,left,mid,ql,qr,val);
            if (qr > mid)  update_change(p<<1|1,mid+1,right,ql,qr,val);
            pushup(p);
        }
    }
    void build(vector<int> &nums) {
        in_build(1,1,n,nums);
    }
    void submit(vector<int> &nums) {
        submit_change(1,1,n,nums);
    }
    void add(int ql,int qr,int val) {
        update_add(1,1,n,ql,qr,val);
    }
    void change(int ql, int qr, int val) {
        update_change(1,1,n,ql,qr,val);
    }
};

void solve()
{
    int q=read();
    vector<array<int,3>> qry(q);
    vector<vector<int>> adj(2);
    int all=1;
    for(int i=0;i<q;++i) {
        int opt=read();
        if(opt==1) {
            int idx=read();
            adj.push_back({});
            adj[idx].push_back(++all);
            qry[i]={opt,idx,all};
        } else { 
            int idx=read(),val=read();
            qry[i]={opt,idx,val};
        }
    }

    vector<int> in(all+1),out(all+1);
    int tim=0;
    auto dfs = [&](auto self,int x) -> void {
        in[x]=++tim;
        for(auto y:adj[x]) {
            self(self,y);
        }
        out[x]=tim;
    };
    dfs(dfs,1);

    SegmentTree seg(all+1);
    vector<int> res(all+1,0);
    seg.build(res);
    for(int i=0;i<q;++i) {
        auto [opt,idx,val]=qry[i];
        if(opt==1) {
            seg.change(in[val],out[val],0);
        } else {
            seg.add(in[idx],out[idx],val);
        }
    }
    seg.submit(res);
    for(int i=1;i<=all;++i) {
        printf("%lld ",res[in[i]]);
    }
    puts("");
}


signed main()
{
    fre(test);
    int T=read();
    while(T--) {
        solve();
        //fflush(stdout);
    }
    return 0;
}
